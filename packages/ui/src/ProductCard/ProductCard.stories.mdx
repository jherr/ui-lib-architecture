import { Meta, Story, Canvas } from '@storybook/addon-docs';

import { ProductCard } from '.';

<Meta title="ProductCard" component={ProductCard} />

export const Template = (args) => <ProductCard {...args } />

# ProductCard Stories

Let's define a story for our `ProductCard` component:

<Story
  name="basic"
  args={{
    title: 'Product Card',
    description: 'A product card',
  }}>
  {Template.bind({})}
</Story>

We can drop it in a `Canvas` to get a code snippet:

<Canvas>
  <Story 
    name="different title"
    args={{
      title: 'Another product',
      description: 'Cool product info',
    }}>
    {Template.bind({})}
  </Story>
</Canvas>
